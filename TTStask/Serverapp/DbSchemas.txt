CREATE TABLE "Adresses" (
	"Id"	INTEGER NOT NULL,
	"Address"	TEXT NOT NULL,
	"Type"	TEXT NOT NULL,
	CONSTRAINT "PK_Adresses" PRIMARY KEY("Id" AUTOINCREMENT)
);
CREATE TABLE "Mouseevents" (
	"Id"	INTEGER NOT NULL,
	"Type"	TEXT NOT NULL,
	"XCoord"	REAL NOT NULL,
	"YCoord"	REAL NOT NULL,
	"dateTime"	TEXT NOT NULL,
	CONSTRAINT "PK_Mouseevents" PRIMARY KEY("Id" AUTOINCREMENT)
);
CREATE TABLE "Users" (
	"Id"	INTEGER NOT NULL,
	"Login"	TEXT NOT NULL,
	"Password"	TEXT NOT NULL,
	"Type"	TEXT NOT NULL,
	CONSTRAINT "PK_Users" PRIMARY KEY("Id" AUTOINCREMENT)
);
